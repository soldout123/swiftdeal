// Document Upload Handler
let uploadedDocs = {
    vehiclePhoto: null,
    registration: null,
    govId: null,
    selfie: null
};

function handleDocUpload(event, docType) {
    const file = event.target.files[0];
    if (!file) return;

    const reader = new FileReader();
    reader.onload = function(e) {
        uploadedDocs[docType] = {
            name: file.name,
            type: file.type,
            data: e.target.result,
            uploadedAt: Date.now()
        };
        
        // Update UI
        const label = document.getElementById(`${docType}Label`);
        label.classList.add('uploaded');
        label.querySelector('.upload-status').textContent = '✅ Uploaded';
    };
    reader.readAsDataURL(file);
}

// Google Play Subscription
function openGooglePlaySubscription() {
    // Check if Android interface is available
    if (typeof Android !== 'undefined' && Android.purchaseSubscription) {
        Android.purchaseSubscription('swiftdeal_driver_pro');
    } else {
        // Fallback for web testing
        alert('Subscription feature available in the Android app. Opening Google Play...');
        // In production web, redirect to Play Store listing
        window.location.href = 'https://play.google.com/store/apps/details?id=com.apps.swiftdeal';
    }
}

// Called by Android when subscription status changes
function updateSubscriptionStatus(isActive) {
    if (currentDriver) {
        database.ref(`drivers/${currentDriver.id}`).update({
            subscriptionStatus: isActive ? 'active' : 'inactive',
            subscriptionUpdated: Date.now()
        });
    }
}

// Check driver status and show appropriate screen
function checkDriverStatus() {
    if (!currentDriver) return;

    if (currentDriver.status === 'pending') {
        // Show pending approval banner
        document.getElementById('pendingBanner').style.display = 'block';
        document.getElementById('dashboardSection').style.display = 'none';
        return;
    }

    if (currentDriver.status === 'rejected') {
        alert('Your application was not approved. Please contact support for details.');
        return;
    }

    if (currentDriver.status === 'approved') {
        // Check subscription
        if (currentDriver.subscriptionStatus !== 'active') {
            document.getElementById('subscriptionLock').style.display = 'block';
            document.getElementById('dashboardSection').style.display = 'none';
        } else {
            document.getElementById('subscriptionLock').style.display = 'none';
            document.getElementById('dashboardSection').style.display = 'block';
        }
    }
}

async function handleDriverSignup() {

    const name = document.getElementById('signupName').value.trim();
    const email = document.getElementById('signupEmail').value.trim();
    const phone = document.getElementById('signupPhone').value.trim();

    const vehiclePlate = document.getElementById('vehiclePlate').value.trim();
    const vehicleYear = document.getElementById('vehicleYear').value;

    // Validation
    if (!name || !email || !phone || !password || !vehicleMake || 
        !vehicleModel || !vehiclePlate || !vehicleYear) {
        alert('Please fill in all fields');

    }

    // Check all documents uploaded
    if (!uploadedDocs.vehiclePhoto || !uploadedDocs.registration || 
        !uploadedDocs.govId || !uploadedDocs.selfie) {

        return;

        console.error('Signup error:', error);
        alert('Signup failed. Please try again.');

### 2. JavaScript Functions for Admin Panel

```javascript
let currentReviewDriverId = null;

// Load pending drivers
function loadDriverQueue() {
    database.ref('drivers')
        .orderByChild('status')
        .equalTo('pending')
        .on('value', (snapshot) => {
            const drivers = snapshot.val();
            const queueDiv = document.getElementById('driverQueue');
            
            if (!drivers) {
                queueDiv.innerHTML = '<p class="empty-state">No pending applications</p>';
                document.getElementById('pendingCount').textContent = '0';
                return;
            }

            const driverList = Object.entries(drivers);
            document.getElementById('pendingCount').textContent = driverList.length;

            queueDiv.innerHTML = driverList.map(([id, driver]) => `
                <div class="driver-card" onclick="reviewDriver('${id}')">
                    <div class="driver-card-header">
                        <h4>${driver.name}</h4>
                        <span class="pending-badge">⏳ Pending</span>
                    </div>
                    <div class="driver-card-body">
                        <p><strong>Vehicle:</strong> ${driver.vehicleMake} ${driver.vehicleModel}</p>
                        <p><strong>Plate:</strong> ${driver.vehiclePlate}</p>
                        <p><strong>Email:</strong> ${driver.email}</p>
                        <p><strong>Submitted:</strong> ${new Date(driver.createdAt).toLocaleDateString()}</p>
                    </div>
                    <button class="btn-review">Review Application →</button>
                </div>
            `).join('');
        });
}

// Open review modal
function reviewDriver(driverId) {
    currentReviewDriverId = driverId;
    
    database.ref(`drivers/${driverId}`).once('value', (snapshot) => {
        const driver = snapshot.val();
        
        // Fill driver info
        document.getElementById('reviewDriverName').textContent = driver.name;
        document.getElementById('reviewDriverEmail').textContent = driver.email;
        document.getElementById('reviewDriverPhone').textContent = driver.phone;
        document.getElementById('reviewDriverVehicle').textContent = 
            `${driver.vehicleMake} ${driver.vehicleModel} (${driver.vehicleYear})`;
        document.getElementById('reviewDriverPlate').textContent = driver.vehiclePlate;

        // Load document previews
        if (driver.documents) {
            document.getElementById('previewVehiclePhoto').src = driver.documents.vehiclePhoto?.data || '';
            document.getElementById('previewRegistration').src = driver.documents.registration?.data || '';
            document.getElementById('previewGovId').src = driver.documents.govId?.data || '';
            document.getElementById('previewSelfie').src = driver.documents.selfie?.data || '';
        }

        // Show modal
        document.getElementById('reviewModal').style.display = 'block';
    });
}

function closeReviewModal() {
    document.getElementById('reviewModal').style.display = 'none';
    document.getElementById('rejectForm').style.display = 'none';
    currentReviewDriverId = null;
}

// Approve driver
async function approveDriver() {
    if (!currentReviewDriverId) return;

    if (confirm('Approve this driver? They will be able to subscribe and go online.')) {
        try {
            await database.ref(`drivers/${currentReviewDriverId}`).update({
                status: 'approved',
                approvedAt: Date.now(),
                approvedBy: currentAdmin.id  // Assuming you have admin ID
            });

            alert('✅ Driver approved successfully!');
            closeReviewModal();
            
            // TODO: Send notification to driver (email/SMS/push)
            
        } catch (error) {
            console.error('Approval error:', error);
            alert('Failed to approve driver. Please try again.');
        }
    }
}

function showRejectForm() {
    document.getElementById('rejectForm').style.display = 'block';
}

// Reject driver
async function confirmRejectDriver() {
    if (!currentReviewDriverId) return;

    const reason = document.getElementById('rejectionReason').value.trim();
    if (!reason) {
        alert('Please enter a rejection reason');
        return;
    }

    if (confirm('Reject this driver application?')) {
        try {
            await database.ref(`drivers/${currentReviewDriverId}`).update({
                status: 'rejected',
                rejectedAt: Date.now(),
                rejectedBy: currentAdmin.id,
                rejectionReason: reason
            });

            alert('❌ Driver application rejected');
            closeReviewModal();
            
            // TODO: Send notification to driver with reason
            
        } catch (error) {
            console.error('Rejection error:', error);
            alert('Failed to reject driver. Please try again.');
        }
    }
}

// Load stats
function loadAdminStats() {
    const today = new Date().setHours(0, 0, 0, 0);

    // Approved today
    database.ref('drivers')
        .orderByChild('approvedAt')
        .startAt(today)
        .on('value', (snapshot) => {
            const count = snapshot.numChildren();
            document.getElementById('approvedToday').textContent = count;
        });

    // Rejected today
    database.ref('drivers')
        .orderByChild('rejectedAt')
        .startAt(today)
        .on('value', (snapshot) => {
            const count = snapshot.numChildren();
            document.getElementById('rejectedToday').textContent = count;
        });
}

// Initialize
document.addEventListener('DOMContentLoaded', () => {
    loadDriverQueue();
    loadAdminStats();
});
```
